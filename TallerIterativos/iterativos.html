import numpy as np
import matplotlib.pyplot as plt

A = np.array([[1, 1],
              [-2, 5]], dtype=float)

b = np.array([7, 0], dtype=float)

D = np.diag(np.diag(A))
R = A - D

def jacobi(x0, it=10):
    x = x0.copy()
    lista = [x.copy()]

    for i in range(it):
        x = np.dot(np.linalg.inv(D), (b - np.dot(R, x)))
        lista.append(x.copy())

    return np.array(lista)

x0a = np.array([0.0, 0.0])
x0b = np.array([5.0, 2.0])

T1 = jacobi(x0a)
T2 = jacobi(x0b)

x = np.linspace(-1, 8, 100)
plt.plot(x, 7 - x)
plt.plot(x, (2/5)*x)

plt.plot(T1[:,0], T1[:,1], 'o-')
plt.plot(T2[:,0], T2[:,1], 's-')

plt.xlabel("x1")
plt.ylabel("x2")
plt.grid()
plt.show()

import numpy as np
import matplotlib.pyplot as plt

A = np.array([[1, 1],
              [-2, 1]], dtype=float)

b = np.array([6, 0], dtype=float)

D = np.diag(np.diag(A))
R = A - D

def jacobi(x0, it=8):
    x = x0.copy()
    puntos = [x.copy()]

    for i in range(it):
        x = np.dot(np.linalg.inv(D), (b - np.dot(R, x)))
        puntos.append(x.copy())

    return np.array(puntos)

x01 = np.array([0.0, 0.0])
x02 = np.array([1.0, 1.0])
x03 = np.array([3.0, 2.0])

T1 = jacobi(x01)
T2 = jacobi(x02)
T3 = jacobi(x03)

x = np.linspace(-20, 20, 200)
plt.plot(x, 6 - x)
plt.plot(x, 2*x)

plt.plot(T1[:,0], T1[:,1], 'o-')
plt.plot(T2[:,0], T2[:,1], 's-')
plt.plot(T3[:,0], T3[:,1], '^-')

plt.xlabel("x1")
plt.ylabel("x2")
plt.grid()
plt.show()
