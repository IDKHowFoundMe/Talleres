def ajustar_min_cuadrados(
    xs: list,
    ys: list,
    gradiente: list[Callable[[list[float], list[float]], tuple]],
) -> np.ndarray:
    """Resuelve el sistema de ecuaciones para encontrar los parmetros del mtodo de mnimos cuadrados. Plantea el sistema de ecuaciones lineales al reemplazar los valores de ``xs`` y ``ys`` en las derivadas parciales.

    ## Parameters

    ``xs``: lista con los valores de x.

    ``ys``: lista con los valores de y.

    ``gradiente``: lista con las funciones de derivadas parciales de ``fcn``.
    [IMPORTANTE] Cada funcin debe ser de la forma ``der_parcial(xi, yi)->``.

    ## Return

    ``solucion``: vector con la solucin del sistema de ecuaciones lineales.

    """

    assert len(xs) == len(ys), "xs y ys deben tener la misma longitud."

    n = len(xs)

    num_pars = len(gradiente)
    logging.info(f"Se ajustarn {num_pars} parmetros.")
    # --- construir matriz A y vector b
    Ab = np.zeros((num_pars, num_pars + 1), dtype=float)

    for i, der_parcial in enumerate(gradiente):
        assert callable(der_parcial), "Cada derivada parcial debe ser una funcin."

        Ab[i, :] = der_parcial(xs, ys)

    # Solve the linear system using NumPy
    A = Ab[:, :-1]
    b = Ab[:, -1]
    return np.linalg.solve(A, b)

# ####################################################################
def der_parcial_1(xs: list, ys: list) -> tuple[float, float, float]:
    """Retorna los coeficientes de la ecuacin de la derivada parcial con respecto al parmetro 1 al reemplazar los valores ``xs`` y ``ys``. La ecuacin es de la forma:
    c_1 * a_1 + c_0 * a_0 = c_ind


    ## Parameters

    ``xs``: lista de valores de x.

    ``ys``: lista de valores de y.

    ## Return

    ``c_1``: coeficiente del parmetro 1.

    ``c_0``: coeficiente del parmetro 0.

    ``c_ind``: coeficiente del trmino independiente.

    """

    # coeficiente del trmino independiente
    c_ind = sum(ys)

    # coeficiente del parmetro 1
    c_1 = sum(xs)

    # coeficiente del parmetro 0
    c_0 = len(xs)

    return (c_1, c_0, c_ind)


def der_parcial_0(xs: list, ys: list) -> tuple[float, float, float]:
    """Retorna los coeficientes de la ecuacin de la derivada parcial con respecto al parmetro 0 al reemplazar los valores ``xs`` y ``ys``. La ecuacin es de la forma:
    c_1 * a_1 + c_0 * a_0 = c_ind


    ## Parameters

    ``xs``: lista de valores de x.

    ``ys``: lista de valores de y.

    ## Return

    ``c_1``: coeficiente del parmetro 1.

    ``c_0``: coeficiente del parmetro 0.

    ``c_ind``: coeficiente del trmino independiente.

    """
    c_1 = 0
    c_0 = 0
    c_ind = 0
    for xi, yi in zip(xs, ys):
        # coeficiente del trmino independiente
        c_ind += xi * yi

        # coeficiente del parmetro 1
        c_1 += xi * xi

        # coeficiente del parmetro 0
        c_0 += xi

    return (c_1, c_0, c_ind)

def der_parcial_a(xs: list, ys: list) -> tuple[float, float, float, float, float]:

    c_3=0
    c_2=0
    c_1 = 0
    c_0 = 0
    c_ind = 0
    for xi, yi in zip(xs, ys):
        # coeficiente del trmino independiente
        c_ind += (xi**3) * yi

        # coeficiente del parmetro 3
        c_3 += xi**6
        
        # coeficiente del parmetro 2
        c_2 += xi**5
        
        # coeficiente del parmetro 1
        c_1 += xi**4

        # coeficiente del parmetro 0
        c_0 += xi**3

    return (c_3, c_2, c_1, c_0, c_ind)
